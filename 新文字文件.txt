<button onclick="setABRepeat()">A-B Repeat</button>

<script>
    var audio = document.getElementById("myAudio");
    var isABRepeatEnabled = false;
    var startPoint = 0;
    var endPoint = 0;
    
    function setABRepeat() {
        if (!isABRepeatEnabled) {
            startPoint = audio.currentTime;
            isABRepeatEnabled = true;
        } else {
            endPoint = audio.currentTime;
            isABRepeatEnabled = false;
            audio.currentTime = startPoint;
        }
    }

    audio.addEventListener('timeupdate', function() {
        if (isABRepeatEnabled && audio.currentTime >= endPoint) {
            audio.currentTime = startPoint;
        }
    });
</script>
